%------------ Expectations, Moments, and MGFs ---------------
\begin{tikzpicture}
    \node [mybox] (box){%
        \begin{minipage}{0.3\textwidth}
        \begin{align*}
            % Eg(X) &= \sum_X g(x) p_X(x) \\
            % Eg(X) &= \int_X g(x) f_X(x) \\
            % E(ag(X) + b) &= aEg(X) + b \\
            % Var(\alpha X + \beta) &= \alpha^2 Var(X) \\
            % \text{rth Moment} &= EX^r \\
            Var(X) &= E(X-EX)^2 \\
            &= EX^2 - (EX)^2 \\
            &= EX(X-1) + EX - (EX)^2 \\
            M_X(t) &= E(e^{tX}) \\
            Cov(X,Y) &= E(X-E(X))(Y-E(Y)) \\
            &= E(XY) - E(X) E(Y) \\
            Var(aX + bY) &= a^2Var(X) + b^2Var(Y) \\
            & \ \ \ + 2abCov(X,Y) \\
            \text{Correlation: } \rho_{X,Y} &= \frac{Cov(X,Y)}{\sqrt{Var(X)Var(Y)}} \\
            f_{X_{(k)}}(x) &= k \binom{n}{k} f(x) [F(x)]^{k-1}[1-F(x)]^{n-k} \\
            \text{Independence: } & F_{X,Y}(x,y) = F_X(x) F_Y(y), \forall x,y \\
            % & p_{X,Y}(x,y) = p_X(x) p_Y(y), \forall x,y \\
            % & f_{X,Y}(x,y) = f_X(x) f_Y(y), \forall x,y \\
            Cov(X,Y) &= 0, \text{ if } X,Y \text{ independent}
        \end{align*}
        \end{minipage}
